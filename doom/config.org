* Ultra-scroll

Smooth pixel-precise scrolling that handles tall images and high-speed
trackpad/mouse input. Replaces =pixel-scroll-precision-mode=.

#+begin_src elisp
(use-package! ultra-scroll
  :init
  (setq scroll-conservatively 3
        scroll-margin 0)
  :config
  (ultra-scroll-mode 1))
#+end_src

** Smooth keyboard scrolling (C-u / C-d)

=ultra-scroll= only handles mouse/trackpad input. For smooth animated
half-page scrolling with =C-u= / =C-d=, we enable the built-in
=pixel-scroll-precision-interpolate-page= (activated by =ultra-scroll=
under the hood) and remap Evil's scroll commands to the interpolated
variants.

#+begin_src elisp
(setq pixel-scroll-precision-interpolate-page t)

(after! evil
  (define-key evil-normal-state-map (kbd "C-d") #'pixel-scroll-interpolate-down)
  (define-key evil-normal-state-map (kbd "C-u") #'pixel-scroll-interpolate-up)
  (define-key evil-visual-state-map (kbd "C-d") #'pixel-scroll-interpolate-down)
  (define-key evil-visual-state-map (kbd "C-u") #'pixel-scroll-interpolate-up))
#+end_src

* IJKL Navigation Setup

This section remaps the core Vim movement keys from =hjkl= to the =ijkl=
"inverted-T" layout and carefully relocates all displaced keys.

** Keybinding Reference

| Key | Normal          | Visual          | Insert  | Operator          | Motion/Minibuf |
|-----+-----------------+-----------------+---------+-------------------+----------------|
| i   | Up              | Extend up       | type i  | (text obj: inner) | type i         |
| k   | Down            | Extend down     | type k  | (unchanged)       | type k         |
| j   | Left            | Extend left     | type j  | (unchanged)       | type j         |
| l   | Right           | Extend right    | type l  | (unchanged)       | type l         |
| ;   | Enter insert    | —               | type ;  | —                 | type ;         |
| :   | Insert at BOL   | —               | type :  | —                 | type :         |
| K   | Join lines      | Join lines      | —       | —                 | —              |
| u   | Undo            | Undo            | —       | —                 | —              |
| U   | Redo            | Redo            | —       | —                 | —              |
| M-; | Ex command      | Ex command      | —       | —                 | —              |

** Core Navigation (Normal, Visual)

Movement keys are bound in =normal= and =visual= states only.
- *Operator-pending* is excluded so text objects (=ciw=, =di(=, =yi"=) work.
- *Motion state* is excluded so minibuffer/M-x/Vertico typing works normally.

#+begin_src elisp
(after! evil
  (defun +custom/apply-ijkl-core-bindings ()

    ;; Core movement in normal + visual states.
    (define-key evil-normal-state-map (kbd "i") #'evil-previous-line)
    (define-key evil-normal-state-map (kbd "k") #'evil-next-line)
    (define-key evil-normal-state-map (kbd "j") #'evil-backward-char)
    (define-key evil-normal-state-map (kbd "l") #'evil-forward-char)
    (define-key evil-visual-state-map (kbd "i") #'evil-previous-line)
    (define-key evil-visual-state-map (kbd "k") #'evil-next-line)
    (define-key evil-visual-state-map (kbd "j") #'evil-backward-char)
    (define-key evil-visual-state-map (kbd "l") #'evil-forward-char)

    ;; Displaced keys.
    (define-key evil-normal-state-map (kbd ";") #'evil-insert)
    (define-key evil-normal-state-map (kbd ":") #'evil-insert-line)
    (define-key evil-normal-state-map (kbd "K") #'evil-join)
    (define-key evil-visual-state-map (kbd "K") #'evil-join)
    (define-key evil-normal-state-map (kbd "U") #'evil-redo)
    (define-key evil-visual-state-map (kbd "U") #'evil-redo)
    (define-key evil-normal-state-map (kbd "M-;") #'evil-ex)
    (define-key evil-visual-state-map (kbd "M-;") #'evil-ex))

  ;; Apply once now, then re-apply after Doom modules load to win precedence.
  (+custom/apply-ijkl-core-bindings)
  (add-hook 'doom-after-modules-config-hook #'+custom/apply-ijkl-core-bindings))
#+end_src

** Window Management (SPC w / C-w)

Doom binds =SPC w= to =evil-window-map= by default, which contains all the
standard window operations (split, close, resize, rotate, etc.). We add our
ijkl navigation directly into that existing map so nothing is lost.

#+begin_src elisp
(defun +custom/window-top-right ()
  "Select the top-right window in the current frame."
  (interactive)
  (let ((best (selected-window))
        (best-right -1))
    (walk-windows
     (lambda (w)
       (let ((edges (window-edges w)))
         (when (and (= (nth 1 edges) 0)            ;; top row
                    (> (nth 2 edges) best-right))   ;; rightmost
           (setq best-right (nth 2 edges)
                 best w))))
     nil (selected-frame))
    (select-window best)))

(map! :map evil-window-map
      "i" #'evil-window-up
      "k" #'evil-window-down
      "j" #'evil-window-left
      "l" #'evil-window-right
      "I" #'+evil/window-move-up
      "K" #'+evil/window-move-down
      "J" #'+evil/window-move-left
      "L" #'+evil/window-move-right
      "t" #'+custom/window-top-right)
#+end_src

=evil-org= adds text-object bindings on =i= in visual state (inner-object,

* Eglot: Keep JSON files in json-mode

Doom's tree-sitter module remaps =json-mode= to =json-ts-mode=.  Because
=json-ts-mode= inherits from =json-mode= which derives from =js-mode=, eglot's
already-running =typescript-language-server= claims JSON buffers and produces
bogus =; expected= diagnostics.  Undo the remap so =.json= files stay in
=json-mode= where eglot correctly uses =vscode-json-language-server=.

#+begin_src elisp
(after! json-ts-mode
  (setf (alist-get 'json-mode major-mode-remap-defaults) nil))
#+end_src

* Linting and Formatting (Biome / Prettier auto-detection)

When a project contains =biome.json=, use Biome for formatting and linting.
Otherwise fall back to Apheleia's default (Prettier).  This means you never
need to change your config when switching between projects.

** Dynamic formatter selection (Apheleia)

The buffer-local =apheleia-formatter= (aliased to =+format-with= in Doom) is
checked *first* by Apheleia, before the mode-alist.  Setting it in a mode hook
gives us per-project formatter selection without touching global state.

#+begin_src elisp
(defun +detect-project-formatter-h ()
  "Set buffer-local formatter to biome when biome.json is found in the project."
  (when (locate-dominating-file default-directory "biome.json")
    (setq-local apheleia-formatter 'biome)))

(add-hook 'typescript-ts-mode-hook #'+detect-project-formatter-h)
(add-hook 'tsx-ts-mode-hook        #'+detect-project-formatter-h)
(add-hook 'js-ts-mode-hook         #'+detect-project-formatter-h)
(add-hook 'json-ts-mode-hook       #'+detect-project-formatter-h)
(add-hook 'jsonc-mode-hook         #'+detect-project-formatter-h)
#+end_src

*** Open opencode on a terminal on the right
#+begin_src elisp
(defun +custom/open-code-terminal ()
  "Open opencode in a new right-most split."
  (interactive)
  ;; Find the right-most window in the current frame.
  (let* ((rightmost-window
          (let ((best (selected-window))
                (best-right -1))
            (dolist (win (window-list nil 'nomini) best)
              (let ((right (nth 2 (window-edges win))))
                (when (> right best-right)
                  (setq best-right right
                        best win))))))
         (target-width 100)
         (evil-vsplit-window-right t))
    (select-window rightmost-window)
    ;; 1. Split at the right-most edge.
    (evil-window-vsplit)
    ;; 3. Resize the new window to a fixed width.
    (window-resize (selected-window)
                   (- target-width (window-total-width))
                   t)
    ;; 4. Open vterm in this new window.
    (+vterm/here nil)
    ;; 5. Start opencode.
    (vterm-send-string "opencode")
    (vterm-send-return)))

;; Now, let's bind it to a Leader key.
;; Since it's a 'code' related terminal, 'SPC o c' (Open Code) is a good fit.
(map! :leader
      (:prefix ("o" . "open")
       :desc "Open Code Terminal" "c" #'+custom/open-code-terminal))
#+end_src
